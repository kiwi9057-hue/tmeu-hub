    <script>
        // --- Data State ---
        let inventory = [
            { id: '1', model: 'Pro Display XDR', partNo: 'MW632TA/A', rackNo: 'A1', position: '04-12', qty: 25, whDate: '2024-01-15' },
            { id: '2', model: 'Mac Studio M2', partNo: 'MQH73TA/A', rackNo: 'B2', position: '01-05', qty: 12, whDate: '2024-02-10' }
        ];
        let history = [];

        // --- View Switching ---
        function switchView(viewId) {
            document.querySelectorAll('.view-section').forEach(s => s.classList.add('hidden'));
            document.getElementById(`view-${viewId}`).classList.remove('hidden');
            
            document.querySelectorAll('.tab-button').forEach(b => b.classList.remove('active'));
            document.getElementById(`tab-${viewId}`).classList.add('active');

            if (viewId === 'inventory') renderInventory();
            if (viewId === 'shipment') updateShipmentDropdown();
            if (viewId === 'history') renderHistory();
        }

        function toggleForm() {
            const card = document.getElementById('inventory-form-card');
            card.classList.toggle('hidden');
        }

        // --- Inventory Logic ---
        document.getElementById('inventory-form').addEventListener('submit', (e) => {
            e.preventDefault();
            const newItem = {
                id: Date.now().toString(),
                model: document.getElementById('model').value,
                partNo: document.getElementById('partNo').value,
                rackNo: document.getElementById('rackNo').value,
                position: document.getElementById('position').value,
                qty: parseInt(document.getElementById('qty').value),
                whDate: document.getElementById('whDate').value
            };
            inventory.push(newItem);
            renderInventory();
            e.target.reset();
            toggleForm();
        });

        function renderInventory() {
            const tbody = document.getElementById('inventory-tbody');
            tbody.innerHTML = inventory.map(item => `
                <tr class="hover:bg-slate-50 transition">
                    <td class="px-4 py-4 font-medium text-slate-900">${item.model}</td>
                    <td class="px-4 py-4 text-slate-500">${item.partNo}</td>
                    <td class="px-4 py-4">
                        <span class="bg-amber-50 text-amber-700 px-2 py-1 rounded text-xs font-bold mr-1">${item.rackNo}</span>
                        <span class="text-slate-400">/</span>
                        <span class="ml-1">${item.position}</span>
                    </td>
                    <td class="px-4 py-4 font-bold ${item.qty < 5 ? 'text-red-500' : 'text-emerald-600'}">${item.qty}</td>
                    <td class="px-4 py-4 text-slate-500">${item.whDate}</td>
                    <td class="px-4 py-4 text-center">
                        <button onclick="deleteItem('${item.id}')" class="text-slate-300 hover:text-red-500 transition">
                            <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" /></svg>
                        </button>
                    </td>
                </tr>
            `).join('');
        }

        function deleteItem(id) {
            if (confirm('ç¢ºå®šè¦ç§»é™¤æ­¤ç­†åº«å­˜è³‡æ–™å—ï¼Ÿ')) {
                inventory = inventory.filter(i => i.id !== id);
                renderInventory();
            }
        }

        // --- Shipment Logic ---
        function updateShipmentDropdown() {
            const select = document.getElementById('ship-item-id');
            select.innerHTML = '<option value="">-- è«‹é¸æ“‡åº«å­˜å“é … --</option>' + 
                inventory.map(item => `<option value="${item.id}">${item.model} (${item.partNo}) - Stock: ${item.qty}</option>`).join('');
        }

        document.getElementById('shipment-form').addEventListener('submit', (e) => {
            e.preventDefault();
            const itemId = document.getElementById('ship-item-id').value;
            const shipQty = parseInt(document.getElementById('ship-qty').value);
            const item = inventory.find(i => i.id === itemId);

            if (item.qty < shipQty) {
                alert('éŒ¯èª¤ï¼šå‡ºè²¨æ•¸é‡å¤§æ–¼ç¾æœ‰åº«å­˜ï¼');
                return;
            }

            // Update Stock
            item.qty -= shipQty;

            // Add History
            history.unshift({
                shipDate: document.getElementById('ship-date').value,
                model: item.model,
                partNo: item.partNo,
                qty: shipQty,
                invoice: document.getElementById('ship-invoice').value,
                customer: document.getElementById('ship-customer').value,
                whDate: item.whDate
            });

            alert('å‡ºè²¨æˆåŠŸï¼åº«å­˜å·²æ‰£é™¤ã€‚');
            e.target.reset();
            switchView('inventory');
        });

        function renderHistory() {
            const tbody = document.getElementById('history-tbody');
            if (history.length === 0) {
                tbody.innerHTML = '<tr><td colspan="6" class="p-8 text-center text-slate-400">ç›®å‰å°šç„¡å‡ºè²¨ç´€éŒ„</td></tr>';
                return;
            }
            tbody.innerHTML = history.map(h => `
                <tr class="hover:bg-slate-50 transition">
                    <td class="px-4 py-4 font-semibold text-sky-800">${h.shipDate}</td>
                    <td class="px-4 py-4">
                        <div class="font-medium">${h.model}</div>
                        <div class="text-xs text-slate-400">${h.partNo}</div>
                    </td>
                    <td class="px-4 py-4 font-bold">-${h.qty}</td>
                    <td class="px-4 py-4"><span class="bg-slate-100 px-2 py-1 rounded text-xs">${h.invoice}</span></td>
                    <td class="px-4 py-4">${h.customer}</td>
                    <td class="px-4 py-4 text-slate-400 text-xs">${h.whDate}</td>
                </tr>
            `).join('');
        }

        // --- Init ---
        window.onload = () => {
            renderInventory();
        };
